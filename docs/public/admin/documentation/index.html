type CORSConfig

CORSConfig содержит настройки для Cross-Origin Resource Sharing (CORS). Определяет разрешенные origins, методы, заголовки, credentials и exposed headers.

type CORSConfig struct {
    AllowOrigins     string
    AllowMethods     string
    AllowHeaders     string
    AllowCredentials bool
    ExposeHeaders    string
}

type DatabaseConfig

DatabaseConfig содержит настройки подключения к базе данных PostgreSQL. Включает параметры хоста, порта, пользователя, пароля, имени базы данных, режима SSL и размеров пула соединений.

type DatabaseConfig struct {
    Host        string
    Port        int
    User        string
    Password    string
    Name        string
    SSLMode     string
    MinPoolSize int
    MaxPoolSize int
}

func (*DatabaseConfig) URL

func (d *DatabaseConfig) URL() string

URL возвращает строку подключения к базе данных в формате PostgreSQL DSN. Формирует URL на основе полей структуры, включая параметры sslmode.

type KeycloakConfig

KeycloakConfig содержит настройки для интеграции с Keycloak. Включает URL issuer, audience, JWKS URL, client ID, secret и имя приложения.

type KeycloakConfig struct {
    IssuerURL    string
    Audience     string
    JWKSURL      string
    ClientID     string
    ClientSecret string

type MinioConfig

MinioConfig содержит настройки для подключения к MinIO (S3-compatible storage). Включает endpoint, ключи доступа, имя bucket, флаг SSL и публичный URL.

type MinioConfig struct {
    Endpoint  string
    AccessKey string
    SecretKey string
    Bucket    string
    UseSSL    bool
    PublicURL string
}

type OTelConfig

OTelConfig содержит настройки для OpenTelemetry. Включает endpoint для экспорта, имя сервиса, протокол и флаг включения.

type OTelConfig struct {
    Endpoint    string
    ServiceName string
    Protocol    string

}

type ServerConfig

ServerConfig содержит настройки сервера. Включает адрес прослушивания, имя приложения и корневой путь API.

type ServerConfig struct {

    AppName  string

}

type Settings

Settings объединяет все конфигурационные структуры в одну. Содержит настройки базы данных, OTel, Keycloak, CORS, сервера, MinIO, тестового модуля и флаг отладки.

type Settings struct {
    Database   DatabaseConfig
    OTel       OTelConfig
    Keycloak   KeycloakConfig
    CORS       CORSConfig
    Server     ServerConfig
    Debug      bool
    Minio      MinioConfig
    TestModule TestModuleConfig
}

func NewSettings

func NewSettings() *Settings

NewSettings создает новый экземпляр Settings, загружая конфигурацию из переменных окружения. Использует вспомогательные функции для загрузки каждой части конфигурации.

func (*Settings) DatabaseURL

func (s *Settings) DatabaseURL() string

DatabaseURL возвращает строку подключения к базе данных, используя метод URL() структуры DatabaseConfig. Это удобный метод для получения полного DSN.

func (*Settings) GetCORSOrigins

func (s *Settings) GetCORSOrigins() []string

GetCORSOrigins возвращает список разрешенных origins для CORS. Если AllowOrigins равно “*”, возвращает ["*"]; иначе разбивает строку по запятым и удаляет пробелы.

func (*Settings) Validate

func (s *Settings) Validate() error

Validate проверяет наличие обязательных переменных окружения для базы данных. Возвращает ошибку, если отсутствуют DB_HOST, DB_USER, DB_PASSWORD или DB_NAME (или DATABASE_URL).

func GetDB

func GetDB() *Database

func InitDB

func InitDB(settings *config.Settings) (*Database, error)

InitDB инициализирует пул соединений с базой данных на основе настроек. Создает pgxpool.Pool, проверяет подключение и сохраняет экземпляр в dbInstance. Возвращает ошибку при неудаче парсинга URL, создания пула или пинга базы данных.

func (*Database) Execute

func (db *Database) Execute(ctx context.Context, query string, args ...interface{}) (int64, error)

Execute выполняет запрос, не возвращающий данные (INSERT, UPDATE, DELETE). Возвращает количество затронутых строк.

func (*Database) ExecuteReturning

func (db *Database) ExecuteReturning(ctx context.Context, query string, args ...interface{}) (map[string]interface{}, error)

ExecuteReturning выполняет запрос, возвращающий одну строку (например, INSERT … RETURNING). Возвращает результат как map[string]interface{}.

func (*Database) FetchAll

func (db *Database) FetchAll(ctx context.Context, query string, args ...interface{}) ([]map[string]interface{}, error)

FetchAll выполняет SELECT-запрос, возвращающий несколько строк. Возвращает все строки результата как []map[string]interface{}.

func (*Database) FetchOne

func (db *Database) FetchOne(ctx context.Context, query string, args ...interface{}) (map[string]interface{}, error)

FetchOne выполняет SELECT-запрос, ожидающий одну строку. Возвращает первую строку результата как map[string]interface{} или nil, если строк нет.
